  - name: Detect and Lint Python
    id: py
    run: |
      py_count=$(find scripts -type f -name '*.py' | wc -l || true)
      echo "py_count=$py_count" >> $GITHUB_OUTPUT

  - name: Lint Python with flake8
    if: steps.py.outputs.py_count != '0'
    run: |
      python -m pip install --upgrade pip
      pip install flake8
      flake8 scripts || true

  - name: Detect and Lint Shell
    id: sh
    run: |
      sh_count=$(find scripts -type f -name '*.sh' | wc -l || true)
      echo "sh_count=$sh_count" >> $GITHUB_OUTPUT

  - name: Lint Shell with shellcheck
    if: steps.sh.outputs.sh_count != '0'
    run: |
      sudo apt-get update
      sudo apt-get install -y shellcheck
      find scripts -type f -name '*.sh' -print0 | xargs -0 shellcheck || true

  - name: Lint Markdown
    run: |
      md_count=$(find . -type f -name '*.md' | wc -l || true)
      if [ "$md_count" -gt 0 ]; then
        python -m pip install --upgrade pip
        pip install markdownlint-cli || true
        markdownlint . || true
      fi
